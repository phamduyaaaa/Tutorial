// Ham Tinh
float Tinh(float x, int n){
	float P = 0;
	int S = 0;
	if(n<=10)
	for(int i = 1;i<=n;i++){
	S +=i;
	P += (x-i)/S;
}
    else
    P = 2020*x +sqrt(pow(n,3));
    
return P;
}
